// Helper: Extract hashtags from text
function extractHashtags(text: string): string[] {
  const matches = text.match(/#[\w\u0600-\u06FF]+/g) || [];
  return matches.map(tag => tag.slice(1)); // Remove # prefix
}

// Helper: Convert object to YAML-like frontmatter
function jsonToYaml(obj: any, indent = 0): string {
  const spaces = '  '.repeat(indent);
  let result = '';
  
  for (const [key, value] of Object.entries(obj)) {
    if (value === undefined || value === null) continue;
    
    if (Array.isArray(value)) {
      if (value.length === 0) {
        result += `${spaces}${key}: []\n`;
      } else if (typeof value[0] === 'object') {
        result += `${spaces}${key}:\n`;
        for (const item of value) {
          result += `${spaces}  -\n`;
          result += jsonToYaml(item, indent + 2);
        }
      } else {
        result += `${spaces}${key}:\n`;
        for (const item of value) {
          result += `${spaces}  - "${item}"\n`;
        }
      }
    } else if (typeof value === 'object') {
      result += `${spaces}${key}:\n`;
      result += jsonToYaml(value, indent + 1);
    } else if (typeof value === 'string') {
      // Handle multiline strings and special characters
      if (value.includes('\n') || value.includes(':') || value.includes('#')) {
        result += `${spaces}${key}: |\n`;
        for (const line of value.split('\n')) {
          result += `${spaces}  ${line}\n`;
        }
      } else {
        result += `${spaces}${key}: "${value.replace(/"/g, '\\"')}"\n`;
      }
    } else {
      result += `${spaces}${key}: ${value}\n`;
    }
  }
  
  return result;
}
